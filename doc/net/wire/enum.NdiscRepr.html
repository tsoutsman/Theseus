    RouterSolicit {
        lladdr: Option<RawHardwareAddress>,
    },
    RouterAdvert {
        hop_limit: u8,
        flags: RouterFlags,
        router_lifetime: Duration,
        reachable_time: Duration,
        retrans_time: Duration,
        lladdr: Option<RawHardwareAddress>,
        mtu: Option<u32>,
        prefix_info: Option<PrefixInformation>,
    },

        target_addr: Address,

    },
    NeighborAdvert {

    },

        target_addr: Address,
        dest_addr: Address,
        lladdr: Option<RawHardwareAddress>,
        redirected_hdr: Option<RedirectedHeader<'a>>,
    },
}
